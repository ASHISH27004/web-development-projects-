// UserDAO.java — connects Servlet to MySQL via JDBC @Repository public class UserDAO { @Autowired private DataSource dataSource; // HikariCP pool injected public List<User> findAll() { var sql = "SELECT * FROM users ORDER BY id"; try (var con = dataSource.getConnection(); var ps = con.prepareStatement(sql); var rs = ps.executeQuery()) { List<User> list = new ArrayList<>(); while (rs.next()) list.add(mapRow(rs)); return list; } } public void insert(User u) throws SQLException { var sql = "INSERT INTO users(name,email,domain,exp_years) VALUES(?,?,?,?)"; try (var con = dataSource.getConnection(); var ps = con.prepareStatement(sql)) { ps.setString(1, u.getName()); ps.setString(2, u.getEmail()); ps.setString(3, u.getDomain()); ps.setInt(4, u.getExpYears()); ps.executeUpdate(); } } public void deleteById(long id) throws SQLException { try (var con = dataSource.getConnection(); var ps = con.prepareStatement( "DELETE FROM users WHERE id=?")) { ps.setLong(1, id); ps.executeUpdate(); } } public void updateDomain(long id, String d, int exp) throws SQLException { try (var con = dataSource.getConnection(); var ps = con.prepareStatement( "UPDATE users SET domain=?,exp_years=? WHERE id=?")) { ps.setString(1, d); ps.setInt(2, exp); ps.setLong(3, id); ps.executeUpdate(); } } private User mapRow(ResultSet rs) throws SQLException { return new User(rs.getLong("id"), rs.getString("name"), rs.getString("email"), rs.getString("domain"), rs.getInt("exp_years")); } }
db-schema.sql — Full Database Schema (6 Tables)
-- NexusAI Full Database Schema (SQLite / MySQL) -- TABLE 1: users CREATE TABLE users ( id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT NOT NULL, email TEXT UNIQUE NOT NULL, domain TEXT DEFAULT 'Undecided', exp_years INTEGER DEFAULT 0, is_active INTEGER DEFAULT 1, joined_at TEXT DEFAULT (datetime('now')) ); CREATE INDEX idx_users_email ON users(email); -- TABLE 2: tech_domains CREATE TABLE tech_domains ( id INTEGER PRIMARY KEY, name TEXT NOT NULL, slug TEXT UNIQUE, market_bn REAL, -- USD billions cagr_pct REAL, -- growth % jobs_k INTEGER, -- open roles K demand TEXT ); -- TABLE 3: job_listings CREATE TABLE job_listings ( id INTEGER PRIMARY KEY AUTOINCREMENT, title TEXT NOT NULL, company TEXT NOT NULL, location TEXT, salary_min INTEGER, -- INR lakhs salary_max INTEGER, tech_stack TEXT, domain_id INTEGER REFERENCES tech_domains(id), posted_date TEXT NOT NULL ); CREATE INDEX idx_jobs_domain ON job_listings(domain_id); -- TABLE 4: salary_data CREATE TABLE salary_data ( id INTEGER PRIMARY KEY AUTOINCREMENT, role TEXT NOT NULL, experience_level TEXT NOT NULL, -- Junior/Mid/Senior/Lead min_salary INTEGER, max_salary INTEGER, avg_salary REAL, yoy_growth_pct REAL, year INTEGER NOT NULL, UNIQUE(role, experience_level, year) ); -- TABLE 5: quiz_results CREATE TABLE quiz_results ( id INTEGER PRIMARY KEY AUTOINCREMENT, user_id INTEGER REFERENCES users(id), score INTEGER NOT NULL, total INTEGER NOT NULL DEFAULT 8, grade TEXT, completed_at TEXT DEFAULT (datetime('now')) ); CREATE INDEX idx_quiz_user ON quiz_results(user_id); -- TABLE 6: enrollments CREATE TABLE enrollments ( id INTEGER PRIMARY KEY AUTOINCREMENT, user_id INTEGER NOT NULL REFERENCES users(id), domain TEXT, status TEXT DEFAULT 'active', progress_pct REAL DEFAULT 0, enrolled_at TEXT DEFAULT (datetime('now')), UNIQUE(user_id, domain) ); -- VIEWS CREATE VIEW v_leaderboard AS SELECT u.name, u.domain, q.score, q.grade FROM users u JOIN quiz_results q ON q.user_id = u.id ORDER BY q.score DESC LIMIT 20; CREATE VIEW v_top_salaries AS SELECT role, experience_level, avg_salary, yoy_growth_pct FROM salary_data ORDER BY avg_salary DESC;
